    agent any
    
  stages {
        stage('git checkout') {
            steps {

        }

            }
        }

            steps {
                nexusArtifactUploader artifacts: [

                        classifier: '',

                    ]
                    ],
                        credentialsId: 'Nexus-cred',
                        groupId: 'in.javahome',
                        nexusUrl: '13.201.12.255:8081',
                        nexusVersion: 'nexus3',
                        protocol: 'http',
                        repository: 'chinna-app',
                        version: '0.1'
        }
            }
  }
