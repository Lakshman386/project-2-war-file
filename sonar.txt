pipeline {
    agent any
    
  stages {
        stage('git checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/Lakshman386/chinna-app.git'
            }
        }
        stage('Generate sonarqube-analysis') {
            steps {
                withSonarQubeEnv(installationName: 'sonarqube', credentialsId: 'sonar-token') {
                    sh 'mvn sonar:sonar'
            }
        }
    }
